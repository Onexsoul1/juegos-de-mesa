import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';

/**
 * Componente principal de la página de inicio.
 * Se encarga de gestionar la barra de navegación según el estado de la sesión almacenada.
 */
@Component({
  selector: 'app-index',
  standalone: true,
  imports: [],
  templateUrl: './index.html',
  styleUrls: ['./index.css']
})
export class IndexComponent implements OnInit {

  /**
   * Constructor del componente.
   * @param router Servicio de enrutamiento para navegar entre rutas.
   */
  constructor(private router: Router) {}

  /**
   * Método que se ejecuta al inicializar el componente.
   * Actualiza la barra de navegación según el usuario autenticado.
   */
  ngOnInit(): void {
    this.actualizarNavbar();
  }

  /**
   * Actualiza la visibilidad de los elementos del navbar en base
   * a la sesión almacenada en localStorage.
   */
  actualizarNavbar(): void {
    const sesionRaw = localStorage.getItem('sesion');
    const navAdmin = document.getElementById('nav-admin');
    const navUser = document.getElementById('nav-user');
    const navLogin = document.getElementById('nav-login');
    const navLogout = document.getElementById('nav-logout');
    const usuarioHeader = document.getElementById('usuario-header');

    if (sesionRaw) {
      const sesion = JSON.parse(sesionRaw);
      if (navLogin) navLogin.classList.add('d-none');
      if (navLogout) navLogout.classList.remove('d-none');
      if (navUser && usuarioHeader) {
        navUser.classList.remove('d-none');
        usuarioHeader.textContent = `👤 ${sesion.usuario}`;
      }
      if (sesion.rol === 'admin' && navAdmin) {
        navAdmin.classList.remove('d-none');
      }
    }
  }

  /**
   * Cierra la sesión activa eliminando la información del almacenamiento local
   * y redirige al usuario al inicio de sesión.
   */
  cerrarSesion(): void {
    localStorage.removeItem('sesion');
    this.router.navigate(['/login']);
  }
}
